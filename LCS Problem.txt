#include<stdio.h>
#include<string.h>

int A[20][20];
int B[20][20];

int LCS_length(char S[],char T[])
{
    int m,n,i,j;
    m = strlen(S);
    n = strlen(T);
    for(i=0; i<=m; i++)
    {
        for(j=0; j<=n; j++)
        {
            if(i==0 || j==0)
            {
                A[i][j]=0;
            }
            else
            {
                if (S[i-1]==T[j-1])
                {
                    A[i][j]=A[i-1][j-1]+1;
                }
                else
                {
                    if(A[i-1][j]>=A[i][j-1])
                    {
                        A[i][j] = A[i-1][j];
                    }
                    else
                    {
                        A[i][j] = A[i][j-1];
                    }
                }

            }
        }
    }
    return A[m][n];
}

int remainder(char S[],char T[])
{
    int m,n,i,j;
    m = strlen(S);
    n = strlen(T);
    for(i=0; i<=m; i++)
    {
        for(j=0; j<=n; j++)
        {
            if(i==0 || j==0)
            {
                B[i][j] = 1;
            }
            else
            {
                B[i][j] = 0;
                if(S[i-1]==T[j-1])
                {
                    B[i][j] = B[i-1][j-1];
                }
                else
                {
                    if(A[i-1][j]==A[i][j])
                    {
                        B[i][j] = (A[i-1][j]+A[i][j])%23102009;
                    }
                    if(A[i][j-1]==A[i][j])
                    {
                        B[i][j] = (A[i][j-1]+A[i][j])%23102009;
                    }
                    if(A[i-1][j-1]==A[i][j])
                    {
                        if(B[i-1][j-1]>B[i][j])
                        {
                            B[i][j] = B[i][j]+23102009;
                            B[i][j]=(B[i][j]-B[i-1][j-1]);
                        }
                    }
                }
            }
        }
    }
    return B[m][n];
}

int main()
{
    int test,p,q;
    char S[20],T[20];
    scanf("%d",&test);
    while(test--)
    {
        scanf("%s%s",S,T);
        p = LCS_length(S,T);
        q = remainder(S,T);
        printf("%d %d\n",p,q);
    }
}
